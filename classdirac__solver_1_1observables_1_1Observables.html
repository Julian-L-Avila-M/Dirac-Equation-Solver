donde H es el Hamiltoniano de Dirac.
H = -iħc * α·∇ + βmc² + V(r)
139 def expected_value_energy(self, psi):
140 """
141 Calcula el valor esperado de la energía: <E> = ∫ ψ† * H * ψ dV
142 donde H es el Hamiltoniano de Dirac.
143 H = -iħc * α·∇ + βmc² + V(r)
144 """
145 dV = self.grid.get_cell_volume()
146
147 # --- Término cinético: -iħc * α·∇ ---
148 grad_psi = self.grid.gradient(psi) # (N_points, dim, 4)
149
150 kinetic_term = np.zeros_like(psi)
151 for k in range(self.grid.dim):
152 alpha_k = dm.alpha[k]
153 term = np.einsum('jk,ik->ij', alpha_k, grad_psi[:, k, :], optimize=True)
154 kinetic_term += term
155
156 kinetic_term *= -1j # Asumimos ħ=c=1
157
158 # --- Término de masa: βmc² ---
159 mass = self.solver.problem.initial_state.mass
160 mass_term = np.einsum('jk,ik->ij', dm.beta, psi, optimize=True) * mass # m=1
161
162 # --- Término de potencial: V(r) ---
163 V = self.solver.problem.potential
164 potential_term = np.zeros_like(psi)
165 if V is not None:
166 # V_grid debe tener forma (N_points,)
167 # Aplanamos coords para iterar sobre cada punto
168 coords_flat = self.grid.coords.reshape(-1, self.grid.dim)
169 V_grid = np.array([V.evaluate(r) for r in coords_flat])
170 # Multiplicar V en cada componente del espinor
171 potential_term = (V_grid[:, np.newaxis] * psi)
172
173 # --- Hamiltoniano aplicado a psi: Hψ ---
174 H_psi = kinetic_term + mass_term + potential_term
175
176 # --- Valor esperado: ∫ ψ† * Hψ dV ---
177 integrand = np.sum(np.conj(psi) * H_psi, axis=1)
178
179 return np.sum(integrand).real * dV

81 def probability_current(self, psi):
82 """
83 Calcula la corriente de probabilidad: J^k = c * ψ† * α^k * ψ
84 donde α^k = γ^0 * γ^k.
85 """
86 # J^k = ψ† * α^k * ψ
87 # Asumimos c=1
88 current = np.zeros((self.grid.get_total_points(), self.grid.dim))
89
90 for k in range(self.grid.dim): # 0=x, 1=y, 2=z
91 # α_k = γ_0 * γ_k
92 alpha_k = dm.alpha[k]
93
94 # (ψ† * α^k * ψ)_i = sum_j,l (ψ†_j * α^k_jl * ψ_l)
95 # Esto se puede hacer con np.einsum para eficiencia
96 term = np.einsum('ij,jk,ik->i', np.conj(psi), alpha_k, psi, optimize=True)
97 current[:, k] = term.real
98
99 return current
