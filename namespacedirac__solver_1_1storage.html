222 """
223 Crea una animación para una simulación 1D, mostrando la densidad de probabilidad

226 with h5py.File(storage_path, 'r') as f:
227 # --- Cargar datos de la simulación ---
228 psi_dset = f['psi']
229 time_dset = f['time']
230
231 # --- Cargar datos de observables ---
232 expected_pos_dset = f['observables/expected_position']
233
234 # --- Cargar metadatos de la malla ---
235 grid_shape = f.attrs['grid_shape']
236 grid_spacing = f.attrs['grid_spacing']
237 grid_origin = f.attrs['grid_origin']
238
239 # Crear coordenadas espaciales para el eje x
240 x_coords = np.linspace(grid_origin[0],
241 grid_origin[0] + (grid_shape[0] - 1) * grid_spacing[0],
242 grid_shape[0])
243
244 fig, (ax1, ax2) = plt.subplots(2, 1, figsize=(10, 8), gridspec_kw={'height_ratios': [3, 1]})
245 fig.suptitle("Simulación 1D de la Ecuación de Dirac")
246
247 # --- Gráfico 1: Densidad de Probabilidad ---
248 ax1.set_xlabel("Posición (x)")
249 ax1.set_ylabel("Densidad de Probabilidad (ρ)")
250 ax1.grid(True)
251 line, = ax1.plot([], [], lw=2)
252 # Línea vertical para la posición esperada
253 pos_marker, = ax1.plot([], [], 'r--', lw=2, label="Posición Esperada <x>")
254 time_text = ax1.text(0.05, 0.9, '', transform=ax1.transAxes)
255
256 # --- Gráfico 2: Evolución de la Posición Esperada ---
257 ax2.set_xlabel("Tiempo")
258 ax2.set_ylabel("Posición Esperada <x>")

263 def init():
264 # Establecer límites de los ejes
265 rho_max = np.max(np.sum(np.abs(psi_dset[0])**2, axis=1))
266 ax1.set_xlim(x_coords[0], x_coords[-1])
267 ax1.set_ylim(0, rho_max * 1.1)
268
269 ax2.set_xlim(0, time_dset[-1])
270 min_pos, max_pos = np.min(expected_pos_dset), np.max(expected_pos_dset)
271 ax2.set_ylim(min_pos - 0.1, max_pos + 0.1)
272
273 line.set_data([], [])
274 pos_marker.set_data([], [])
275 expected_pos_line.set_data([], [])
276 time_text.set_text('')
277 ax1.legend()
278 return line, pos_marker, time_text, expected_pos_line
279
